                <input type="radio" name="mode" value="balanced" checked>
                <span class="radio-circle">

            </label>

                <input type="radio" name="mode" value="power-saving">
                <span class="radio-circle">

                </span>

        <div id="resultDisplay" class="result-display">
            您当前选择的是：平衡模式
        </div>
    </div>

    <script>
        const radioOptions = document.querySelectorAll('.radio-option input');
        const resultDisplay = document.getElementById('resultDisplay');

        // 初始化显示默认选中的结果
        updateResultDisplay();
        
        // 为每个单选按钮添加事件监听
        radioOptions.forEach(radio => {
            radio.addEventListener('change', function() {
                if(this.checked) {
                    updateResultDisplay();
                }
            });
        });

        // 更新结果显示
        function updateResultDisplay() {
            const selectedOption = document.querySelector('input[name="mode"]:checked');
            let displayText = '';
            
            switch(selectedOption.value) {
                case 'performance':
                    displayText = '高性能模式';
                    break;
                case 'balanced':
                    displayText = '平衡模式';
                    break;
                case 'power-saving':
                    displayText = '节能模式';
                    break;
            }
            
            resultDisplay.textContent = `您当前选择的是：${displayText}`;
        }
    </script>
